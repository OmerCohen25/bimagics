Query Behind the Numbers

SELECT DATE_TRUNC(order_date, MONTH) AS month,
       SUM(total_amount)             AS revenue
FROM   bigquery_project.orders
WHERE  order_date >= DATE_SUB(CURRENT_DATE(), INTERVAL 12 MONTH)
GROUP  BY month
ORDER  BY month;
